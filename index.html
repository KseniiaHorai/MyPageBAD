        if (arr.length <= 1) {
            return arr;
        }

        const pivot = arr[0];
        const less = [];
        const greater = [];

        for (let i = 1; i < arr.length; i++) {
            if (arr[i] > pivot) {
                greater.push(arr[i]);
            } else {
                less.push(arr[i]);
            }
        }

        return [...quickSort(less), pivot, ...quickSort(greater)];
    }

Find the longest increasing and decreasing sequences

    //3. Знайдемо найбільшу послідовність чисел, 
    //яка збільшується і яка зменшується

    function findLongestIncreasingSequence(arr) {
        let maxSeq = [];
        let currentSeq = [];

        for (let i = 0; i < arr.length; i++) {
            if (i === 0 || arr[i] > arr[i - 1]) {
                currentSeq.push(arr[i]);
            } else {
                if (currentSeq.length > maxSeq.length) {
                    maxSeq = currentSeq;
                }
                currentSeq = [arr[i]];
            }
        }

        if (currentSeq.length > maxSeq.length) {
            maxSeq = currentSeq;
        }

        return maxSeq;
    }



    function findLongestDecreasingSequence(arr) {

                }

            }
        }

        }
